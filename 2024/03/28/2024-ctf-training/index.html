def initialize(key):

def generate_key_stream(S, length):
    i = 0
    j = 0
    key_stream = []
    for _ in range(length):
        i = (i + 1) % 256
        j = (j + S[i]) % 256
        S[i], S[j] = S[j], S[i]
        key_stream.append(S[(S[i] + S[j]) % 256])
    return key_stream

def decrypt(data, key):
    S = initialize(key)

去除花指令

由于TLS回调执行早于main，所以优先分析（2023强网杯re签到也是）

进来可以看见TLS，由于命名是3，猜测前面还有2个TLS

可以看见0x00401042,0x00401134爆红

0x00401042可见该指令循环跳转到0x400FD1，而本程序起始地址为0x00401000，不存在0x400FD1，应该把这里的指令nop掉

步骤
  1. 我们应该现在0x401042处按u，然后nop掉0x401042的单个字节（即用change byte把第一个字节改成90）。
  2. 将下面无法识别的data转成code
  3. 在tls入口处按u，取消原来的define，再按p，生成可以反编译的tls函数

这里经过gpt计算，远程调用的地址为0x5F84D0B，显然我们的程序无法到达，所以nop掉，操作和上面相同。

在这个花中，需要把黄框内的字节patch掉

def xtea_decrypt(v, k, r):
    v0 = c_uint32(v[0])
    v1 = c_uint32(v[1])
    delta = 0x9e3779b9
    sum1 = c_uint32(delta * r)
    for i in range(r):
        v1.value -= (((v0.value << 4) ^ (v0.value >> 5)) + v0.value) ^ (sum1.value + k[(sum1.value >> 11) & 3])
        sum1.value -= delta
        v0.value -= (((v1.value << 4) ^ (v1.value >> 5)) + v1.value) ^ (sum1.value + k[sum1.value & 3])
    return v0.value, v1.value

def byte2uint32(d):
    l = []
    for i in range(len(d) // 4):
        tmp = d[i * 4:i * 4 + 4]
        l.append(tmp[3] * 256 ** 3 + tmp[2] * 256 ** 2 + tmp[1] * 256 + tmp[0])
    return l


def uint322byte(uint32_list):
    byte_array = []

        byte_array.append((number >> 8) & 0xFF)
        byte_array.append((number >> 16) & 0xFF)
        byte_array.append((number >> 24) & 0xFF)
    return byte_array

def rc4(data, key):
    S = list(range(256))
    j = 0
    out = []

    for i in range(256):
        j = (j + S[i] + key[i % len(key)]) % 256
        S[i], S[j] = S[j], S[i]

    i = j = 0
    for char in data:
        i = (i + 1) % 256
        j = (j + S[i]) % 256
        S[i], S[j] = S[j], S[i]
        out.append(char ^ S[(S[i] + S[j]) % 256])

    return bytes(out)
