（1）

源程序

assume cs:code,ds:data,ss:stack

data segment
	dw 0123h,0456h,0789h,0abch,0defh,0fedh,0cbah,0987h
data ends
stack segment
	dw 0,0,0,0,0,0,0,0
stack ends

code segment

 start: mov ax,stack
		mov ss,ax
		mov sp,16
		
		mov ax,data
		mov ds,ax
		
		push ds:[0]
		push ds:[2]
		pop ds:[2]
		pop ds:[0]

		mov ax,4c00h
		int 21h
code ends
end start

一直按t，返回前查看寄存器状态

（5）

assume cs:code

a segment
	db 1,2,3,4,5,6,7,8
a ends

b segment
	db 1,2,3,4,5,6,7,8
b ends

c segment
	db 0,0,0,0,0,0,0,0
c ends

code segment
start:
	mov ax,a
	mov es,ax					;附加段
	
	mov ax,c
	mov ds,ax
	
	mov bx,0
	mov cx,8
	
s0:
	mov ax,es:[bx]				;加a段的
	add [bx],ax
	add bx,2
	loop s0
	
	mov ax,b
	mov es,ax
	
	mov bx,0
	mov cx,8
s1:
	mov ax,es:[bx]				;加b段的
	add [bx],ax
	add bx,2
	loop s1
	
	mov ax,4c00h
	int 21h
	
code ends
end start

（6）

直接把b段当作栈结构即可

assume cs:code

a segment
	db 1,2,3,4,5,6,7,8,9,0ah,0bh,0ch,0dh,0eh,0fh
a ends

b segment
	db 0,0,0,0,0,0,0,0
b ends

code segment
start:
	mov ax,a
	mov ds,ax
	
	mov ax,b
	mov ss,ax
	mov sp,16						;把b当成栈结构，指向栈顶
    
    mov bx,0
    mov cx,8						;循环8次
    
s:
	push ds:[bx]
	add bx,2
	loop s
	
	mov ax,4c00h
	int 21h
	
code ends
end start
